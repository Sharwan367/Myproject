What is Full Stack Development?

Full stack means building the frontend (what users see) and the backend (the logic and data). The frontend uses HTML, CSS, and JavaScript. The backend runs on a server, processes requests, and talks to a database.

In Plain Words
  • HTML gives the page its structure (headings, paragraphs, forms).
  • CSS makes it look good (colors, spacing, layout).
  • JavaScript makes it interactive (buttons that do things, data validation).
  • Backend handles requests (e.g., save a form), applies rules, and returns results.
  • Database stores and retrieves data (users, products, posts).
A full stack developer understands how these parts communicate to create a complete application.

  const a = parseFloat(document.getElementById('a').value);
  const b = parseFloat(document.getElementById('b').value);
  const out = document.getElementById('calcResult');
  
  if (op === 'clear') { 
    document.getElementById('a').value = ''; 
    document.getElementById('b').value = ''; 
    out.textContent = 'â€”'; 
    return; 
  }
  
  if (Number.isNaN(a) || Number.isNaN(b)) { 
    out.textContent = 'Enter numbers first.'; 
    return; 
  }
  
  const map = { 
    add: a + b, 
    sub: a - b, 
    mul: a * b, 
    div: b === 0 ? 'Cannot divide by 0' : a / b 
  };
  
  out.textContent = map[op];
}
Todo List (with local storage)
    Show JavaScript code 
    // Todo app functionality
const storeKey = 'todos-v1';

function loadTodos() {
  try {
    return JSON.parse(localStorage.getItem(storeKey)) || [];
  } catch {
    return [];
  }
}

function saveTodos(items) {
  localStorage.setItem(storeKey, JSON.stringify(items));
}

function renderTodos() {
  const list = document.getElementById('todoList');
  list.innerHTML = '';
  
  for (const item of loadTodos()) {
    const li = document.createElement('li');
    li.className = 'list-group-item d-flex align-items-center';
    if (item.done) li.classList.add('completed');
    
    const ck = document.createElement('input');
    ck.type = 'checkbox';
    ck.checked = item.done;
    ck.addEventListener('change', () => {
      item.done = ck.checked;
      update(item);
    });
    
    const span = document.createElement('span');
    span.textContent = item.text;
    span.className = 'flex-grow-1 ms-2';
    
    const del = document.createElement('button');
    del.innerHTML = '×';
    del.className = 'btn btn-sm btn-outline-danger';
    del.addEventListener('click', () => remove(item));
    
    li.append(ck, span, del);
    list.append(li);
  }
}

function add(text) {
  const items = loadTodos();
  items.push({ id: Date.now(), text, done: false });
  saveTodos(items);
  renderTodos();
}

function update(item) {
  const items = loadTodos().map(x => x.id === item.id ? item : x);
  saveTodos(items);
  renderTodos();
}

function remove(item) {
  const items = loadTodos().filter(x => x.id !== item.id);
  saveTodos(items);
  renderTodos();
}
